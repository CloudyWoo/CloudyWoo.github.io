``` R
# 벡터 연산

## 벡터 연산은 원소 대 원소로 연산을 진행한다.
1 + 2
"+"(1, 2)


## 첫 번째 벡터의 각 원소와 두 번째 벡터의 각 원소끼리 연산이 진행된다.
c(1, 2, 3) + c(4, 5, 6)
c(1, 2, 3) - c(4, 5, 6)
c(1, 2, 3) * c(4, 5, 6)
c(1, 2, 3) / c(4, 5, 6)

c(10, 20, 30) / c(2, 4, 6)

## %%: 나눈 다음 나머지를 취하는 연산자
c(10, 20, 30) %% c(3, 5, 7)

## %/%: 나눈 후 몫을 취하는 연산자
c(10, 20, 30) %/% c(3, 5, 7)


# 앞 뒤 벡터의 길이가 다르다면 어떻게 될까?
## 긴 벡터의 원소 중 일부는 대응되는 원소를 갖지 못한다.

## a.k.a. recycling rule

## 짧은 벡터의 첫 원소부터 반복사용
c(1, 2, 3) + c(4, 5, 6, 7, 8, 9)
[1]  5  7  9  8  10  12

c(1, 3, 5) + 10
[1] 11 13 15

# 두 벡터의 원소의 개수가 배수 관계일 경우
"""
두 객체의 길이가 서로 배수관계가 아닐 경우 - 경고 출력
다소 불완전한 방식으로 연산이 이루어짐 - 점검 필요
"""

c(1, 2, 3) + c(4, 5, 6, 7, 8)
"""
Warning message:
In c(1, 2, 3) + c(4, 5, 6, 7, 8) :
  longer object length is not a multiple of shorter object length
"""

## 배수관계인 경우
c(1, 2, 3) + c(4, 5, 6, 7, 8, 9)
[1]  5  7  9  8  10  12
```

``` R
# 논리 연산자 
"""
(==, !=, <, <=, >=, >, !x, x | y, x & y, isTRUE())

- !x : x가 아니다.
- x | y : x 또는 y
- x & y : x 그리고 y
"""

v = pi
w = 10/3

v
[1] 3.141593

w
[1] 3.333333

v == w
[1] FALSE

v != w
[1] TRUE

(v == w) | (v < w)
[1] TRUE

## 논리 연산자도 벡터 연산을 수행한다.
## 대응되는 각 원소를 비교하는 벡터 연산 수행
y = c(0, 25, 50, 75, 100)
z = c(50, 50, 50, 50, 50)

y == z

## 논리값을 산술 연산에 사용할 수 있다.
TRUE * TRUE
[1] 1

TRUE * FALSE
[1] 0

TRUE + TRUE
[1] 2

## 논리 연산자를 이용하여 조건을 만족하는 원소의 개수를 셀 수 있다.
## 벡터 y에서 50보다 큰 원소의 개수
y = c(0, 25, 50, 75, 100)

y > 50
[1] FALSE  FALSE  FALSE  TRUE  TRUE

## 벡터를 비교한 후 일부 또는 전부가 TRUE인지 확인하는 방법이 있다.
any(-3:3 > 0)

[1] TRUE
all(-3:3 > 0)
[1] FALSE

## 주의
## 아래의 경우 당연히 수학적으로 TRUE 값이 나와야 하지만 FALSE가 출력된다.  
## R에선 내부적으로 모든 연산을 부동소수점 연산으로 저장하기 때문에 
## 매우 작은 값이라도 반올림으로 인해 오차가 발생할 수 있다.

sqrt(2)^2 == 2
[1] FALSE

sqrt(2)^2 - 2
[1] 4.440892e-16

identical(sqrt(2)^2, 2) 
[1] FALSE

## 미세한 차이를 무시하고 싶을 때 사용하는 solution!
all.equal(sqrt(2)^2, 2)
[1] TRUE

all.equal(sqrt(2)^2, 3)
[1] "Mean relative difference: 0.5"

isTRUE(all.equal(sqrt(2)^2, 3))
[1] FALSE

fruit = c("Apple", "Banana", "Strawberry")
food = c("Pie", "Juice", "Cake")

isTRUE(all.equal(sqrt(2)^2, 3))
[1] FALSE
```
